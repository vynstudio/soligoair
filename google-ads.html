⚙️ Connect Google Ads — One-Time Setup
Paste this script into Google Ads once. It runs hourly and pushes your data here automatically — no API keys, no OAuth.
1
Open Google Ads Scripts
Go to ads.google.com → Tools & Settings → Bulk Actions → Scripts → click + to create a new script.
2
Paste this script
Copy the entire script below and paste it into the editor. Click Authorize when prompted.
function main() { var ENDPOINT = 'https://soligoair.shop/.netlify/functions/google-ads-sync'; var SECRET = 'soligo-gads-2026'; var DATE_RANGE = 'LAST_30_DAYS'; var campaigns = []; var adgroups = []; var keywords = []; // Campaigns var campIt = AdsApp.campaigns() .withCondition('Status != REMOVED') .withDateRange(DATE_RANGE) .get(); while (campIt.hasNext()) { var c = campIt.next(); var s = c.getStatsFor(DATE_RANGE); campaigns.push({ id: c.getId(), name: c.getName(), status: c.isEnabled() ? 'ACTIVE' : 'PAUSED', type: c.getAdvertisingChannelType(), budget: c.getBudget().getAmount(), spend: s.getCost(), clicks: s.getClicks(), impressions: s.getImpressions(), ctr: s.getCtr(), avgCpc: s.getAverageCpc(), conversions: s.getConversions(), costPerConv: s.getCostPerConversion() }); } // Ad Groups var agIt = AdsApp.adGroups() .withCondition('Status != REMOVED') .withDateRange(DATE_RANGE) .get(); while (agIt.hasNext()) { var ag = agIt.next(); var s = ag.getStatsFor(DATE_RANGE); adgroups.push({ id: ag.getId(), name: ag.getName(), campaignId: ag.getCampaign().getId(), campaign: ag.getCampaign().getName(), status: ag.isEnabled() ? 'ACTIVE' : 'PAUSED', spend: s.getCost(), clicks: s.getClicks(), ctr: s.getCtr(), avgCpc: s.getAverageCpc(), conversions:s.getConversions() }); } // Keywords var kwIt = AdsApp.keywords() .withCondition('Status != REMOVED') .withDateRange(DATE_RANGE) .orderBy('Clicks DESC') .withLimit(100) .get(); while (kwIt.hasNext()) { var kw = kwIt.next(); var s = kw.getStatsFor(DATE_RANGE); keywords.push({ text: kw.getText(), matchType: kw.getMatchType(), campaign: kw.getCampaign().getName(), adGroup: kw.getAdGroup().getName(), status: kw.isEnabled() ? 'ACTIVE' : 'PAUSED', spend: s.getCost(), clicks: s.getClicks(), impressions: s.getImpressions(), ctr: s.getCtr(), avgCpc: s.getAverageCpc(), conversions: s.getConversions(), qualityScore:kw.getQualityScore() }); } var payload = JSON.stringify({ campaigns: campaigns, adgroups: adgroups, keywords: keywords, accountId: '727-865-0950', dateRange: DATE_RANGE }); var options = { method: 'post', contentType: 'application/json', headers: { 'X-Secret': SECRET }, payload: payload, muteHttpExceptions: true }; var resp = UrlFetchApp.fetch(ENDPOINT, options); Logger.log('Sync status: ' + resp.getResponseCode()); }
3
Set hourly schedule
After saving, click Run once to test it. Then click the clock icon → set frequency to Every hour. Done — your data will appear here automatically.
4
Come back here
Refresh this page after running the script. Your campaigns, ad groups, and keywords will appear within seconds.
